 21 // Celem projektu jest wyświetlanie obiektów JavaScript'owych w konsoli
 22 // plain-tekstowej.
 23 //
 24 // Powodem dla którego realizuję ten projekt jest potrzeba wyświetlania, w sposób
 25 // czytelny dla developera, obiektó których chcemy zobaczyć budowę w środowisku
 26 // urzadzeń TV.
 27 //
 28 // Jako przykład niech posłużą urządzenia TV Samsung oraz BluRay.
 29 // Producent udostępnia swoje SDK, ale sposób debugowania opiera się jeszcze
 30 // o konsole w trybie tylko tekstowym czy nie tak jak ma to miejsce w Firebug'u
 31 // - najlepszym narzędziu dla developerów w przeglądarce Mozilla Firefox.
 32 // W najpopularniejszej przeglądarce Chrome istnieje wbudowane narzędzie
 33 // dla developerów o nazwie DevTools. Dzięki temu narzędziu mamy zdecydowanie
 34 // więcej możliwości związanych z debudowanie aplikacji na WWW.
 35 //
 36 // Telewizory oraz BluRay'e nie są (jeszcze) tak popularne aby istniały natywne
 37 // rozwiązania które ulepszą pracę developerom. Dlatego ten projekt ma w swoich
 38 // celach przyjść z pomocą gdy zainstnieje potrzeba wyprintować jakiś obiekt który
 39 // zostanie stowrzony w środowisku silnika WebKit (np. TV Samsung 2012)
 40 
 41 function logger(data) {
 42     var res;
 43 
 44     if (this instanceof logger) {
 45         return logger;
 46     }
 47 
 48     // parse data for get 2 value
 49     // - parse_type - where data is comes
 50     // - parse_data - TODO: for what?
 51     res = logger.parser(data);
 52 
 53     // run concrete logger
 54     return logger[res.parse_type](res.parse_data);
 55 };
 56 
 57 // public API
 58 module.exports = logger;
