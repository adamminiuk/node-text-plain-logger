 24 function logger(data, indent) {
 25     indent = indent || 0;
 26 
 27     if (typeof indent !== "number") {
 28         throw new Error("logger: indent is not number");
 29     }
 30 
 31     var i,
 32         // returned value
 33         res,
 34 
 35         // available special logger types
 36         parts = ["DOMLogger", "JSLogger"],
 37 
 38         // number of special loggers
 39         len = parts.length;
 40 
 41     // reset found status
 42     logger.found = false;
 43 
 44     // check if some special logger found value
 45     for (i = 0; i < len; ++i) {
 46         if ( (res = logger[parts[i]](data, indent)) !== undefined ) {
 47             logger.found = true;
 48             break;
 49         }
 50     }
 51 
 52     if ( logger.found ) {
 53         // if logger model has matched also returned parsing value
 54         return res;
 55     }
 56 
 57     // if not found, report w exception
 58     throw new Error("logger: unexpected data: undefined type of variable: " + logger.JSLogger({
 59         // value convert to string
 60         "toString": Object.prototype.toString.call( data ),
 61         "typeof": typeof data,
 62         "contructor": data.constructor && data.constructor.name
 63     }));
 64 }
 65 
 66 // found status
 67 logger.found = false;
 68 
 69 // parser's
 70 logger.parser = {};
 71 
 72 // public API
 73 if (module !== undefined) {
 74     // only for NodeJS
 75     module.exports = logger;
 76 }
